2. **Add openSettingsWindow() static method** after the existing `showMainWindow()` method:
```swift
static func openSettingsWindow() {
    NSApp.setActivationPolicy(.regular)
    for window in NSApp.windows where window.identifier?.rawValue == "settings" {
        window.makeKeyAndOrderFront(nil)
        NSApp.activate(ignoringOtherApps: true)
        return
    }
    // First time: window not yet instantiated by SwiftUI — activate to trigger creation
    NSApp.activate(ignoringOtherApps: true)
}
```

3. **Add activation policy restoration** — Add an `onReceive` modifier to the Settings Window content that listens for `NSWindow.willCloseNotification` and restores `.accessory` policy when the settings window closes. Use this pattern inside the Window scene content:

```swift
.onReceive(NotificationCenter.default.publisher(for: NSWindow.willCloseNotification)
    .compactMap { $0.object as? NSWindow }
    .filter { $0.identifier?.rawValue == "settings" }
) { _ in
    NSApp.setActivationPolicy(.accessory)
}
```
