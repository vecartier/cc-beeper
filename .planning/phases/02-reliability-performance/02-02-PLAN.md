   ```swift
       // Render once, cache forever — noise is deterministic (seeded RNG)
       private static let cachedImage: NSImage = {
           // Render at a fixed size that covers the shell (186x224 is shellW x shellH)
           let width = 186
           let height = 224
           let step: CGFloat = 1.5
           let img = NSImage(size: NSSize(width: width, height: height))
           img.lockFocus()
           var rng = SeededRNG(seed: 42)
           var y: CGFloat = 0
           while y < CGFloat(height) {
               var x: CGFloat = 0
               while x < CGFloat(width) {
                   let val = rng.next()
                   if val < 0.45 {
                       NSColor.white.withAlphaComponent(val * 0.7).setFill()
                       NSRect(x: x, y: y, width: step, height: step).fill()
                   } else if val > 0.55 {
                       NSColor.black.withAlphaComponent((1.0 - val) * 0.5).setFill()
                       NSRect(x: x, y: y, width: step, height: step).fill()
                   }
                   x += step
               }
               y += step
           }
           img.unlockFocus()
           return img
       }()

       var body: some View {
           Image(nsImage: Self.cachedImage)
               .resizable()
       }
   }
   ```

   This renders the noise texture exactly once at app startup (via lazy static), then reuses the cached NSImage on every frame. The SeededRNG and pixel logic are identical — just moved from per-frame Canvas to one-time NSImage rendering.

   Keep SeededRNG struct as-is (other code might use it, and it's small).

- ContentView.swift lines 429-441: `Color(hex:)` extension handles 6-char hex

**Changes to ThemeManager.swift:**

Rewrite `darken()` to use the existing `Color(hex:)` init instead of duplicating hex parsing. Since Color doesn't expose its RGB components directly, and we need to apply a darkening factor, the cleanest approach is:

1. Add a helper method that parses hex to RGB doubles (extracting the shared logic):

   In the `Color(hex:)` extension in ContentView.swift, add a static helper:
   ```swift
   extension Color {
       /// Parse a hex string into (r, g, b) components in 0...1 range.
       static func hexComponents(_ hex: String) -> (r: Double, g: Double, b: Double) {
           let cleaned = hex.trimmingCharacters(in: CharacterSet.alphanumerics.inverted)
           var int: UInt64 = 0
           Scanner(string: cleaned).scanHexInt64(&int)
           switch cleaned.count {
           case 6:
               return (
                   r: Double(int >> 16) / 255,
                   g: Double(int >> 8 & 0xFF) / 255,
                   b: Double(int & 0xFF) / 255
               )
           default:
               return (r: 0, g: 0, b: 0)
           }
       }

       init(hex: String) {
           let c = Self.hexComponents(hex)
           self.init(.sRGB, red: c.r, green: c.g, blue: c.b)
       }
   }
   ```

2. In ThemeManager.swift, rewrite `darken()` to use `Color.hexComponents()`:
   ```swift
   private static func darken(_ hex: String, by factor: Double) -> Color {
       let c = Color.hexComponents(hex)
       return Color(.sRGB, red: c.r * (1 - factor), green: c.g * (1 - factor), blue: c.b * (1 - factor))
   }
   ```

   This removes all duplicate hex parsing from ThemeManager. The `darken` method no longer contains Scanner, trimmingCharacters, or bit-shifting — it delegates entirely to `Color.hexComponents()`.

3. Remove any import that was only needed for the old parsing (CharacterSet is in Foundation which is already imported via SwiftUI).

The result: one hex parsing implementation (`Color.hexComponents`), used by both `Color(hex:)` and `ThemeManager.darken()`. Any future hex parsing changes only need to happen in one place.
