enum PermissionMode: String, CaseIterable, Equatable {
    case cautious
    case guided
    case guardedYolo
    case fullYolo

        switch self {
        case .cautious: return "default"
        case .guided: return "plan"

        case .guided: return "Guided"
        case .guardedYolo: return "Guarded YOLO"
        case .fullYolo: return "Full YOLO"
        }
    }

    var isYolo: Bool {
        self == .guardedYolo || self == .fullYolo
    }
}

**Step 6: Create PermissionModeWriter.swift**

New file at `Sources/Monitor/PermissionModeWriter.swift`:

```swift
import Foundation

enum PermissionWriteError: LocalizedError {
    case malformedSettings
    case writeFailure(String)

    var errorDescription: String? {
        switch self {
        case .malformedSettings:
            return "settings.json is malformed — cannot read or write permission mode."
        case .writeFailure(let detail):
            return "Failed to write settings.json: \(detail)"
        }
    }
}

enum PermissionModeWriter {
    private static let settingsPath = NSHomeDirectory() + "/.claude/settings.json"
    private static let cachePath = ClaudeMonitor.ipcDir + "/cached-deny-rules.json"

    /// Write permission mode to settings.json. Compact encoding — no .prettyPrinted, no .sortedKeys (per D-03).
    /// For Full YOLO: caches deny rules before clearing them.
    /// For non-Full-YOLO: restores deny rules from cache if switching away from Full YOLO.
    static func writePermissionMode(_ mode: PermissionMode, previousMode: PermissionMode) throws {
        let fm = FileManager.default
        guard let data = fm.contents(atPath: settingsPath),
              var settings = try? JSONSerialization.jsonObject(with: data) as? [String: Any] else {
            throw PermissionWriteError.malformedSettings
        }

        // Cache deny rules before switching TO Full YOLO
        if mode == .fullYolo && previousMode != .fullYolo {
            try cacheDenyRules(from: settings)
        }

        // Set permission_mode
        settings["permission_mode"] = mode.rawSettingsValue

        // Handle deny rules
        if mode == .fullYolo {
            // Clear deny rules
            settings.removeValue(forKey: "deny")
        } else if previousMode == .fullYolo {
            // Restore deny rules from cache
            if let restored = restoreDenyRules() {
                settings["deny"] = restored
            }
            // If no cache, don't add deny — they were already cleared
        }

        // Write compact (no .prettyPrinted, no .sortedKeys — per D-03, spec section 9)
        let newData = try JSONSerialization.data(withJSONObject: settings)
        let tmpPath = settingsPath + ".tmp"
        try newData.write(to: URL(fileURLWithPath: tmpPath), options: .atomic)
        if fm.fileExists(atPath: settingsPath) {
            _ = try fm.replaceItemAt(
                URL(fileURLWithPath: settingsPath),
                withItemAt: URL(fileURLWithPath: tmpPath)
            )
        } else {
            try fm.moveItem(atPath: tmpPath, toPath: settingsPath)
        }
    }

    /// Cache current deny rules to cached-deny-rules.json
    static func cacheDenyRules(from settings: [String: Any]) throws {
        let deny = settings["deny"] as? [[String: Any]] ?? []
        let data = try JSONSerialization.data(withJSONObject: deny)
        try data.write(to: URL(fileURLWithPath: cachePath))
    }

    /// Restore deny rules from cache. Returns nil if cache doesn't exist or is unreadable.
    static func restoreDenyRules() -> [[String: Any]]? {
        guard let data = FileManager.default.contents(atPath: cachePath),
              let deny = try? JSONSerialization.jsonObject(with: data) as? [[String: Any]] else {
            return nil
        }
        return deny
    }

    /// Load permission mode for UI display. Returns (mode, malformed) tuple.
    /// If settings.json is unreadable, returns (.cautious, true).
    static func loadPermissionModeForUI() -> (mode: PermissionMode, malformed: Bool) {
        let fm = FileManager.default
        guard let data = fm.contents(atPath: settingsPath),
              let _ = try? JSONSerialization.jsonObject(with: data) as? [String: Any] else {
            return (.cautious, true)
        }
        return (readPermissionMode(), false)
    }
}
```
